010package org.fusesource.hawtjni.runtime;
011
012/**
013 * 
014 * @author <a href="http://hiramchirino.com">Hiram Chirino</a>
015 */
016public enum FieldFlag {
017    /** 
018     * Indicate that the item should not be generated. For example, 
019     * custom natives are coded by hand. 
020     */
021    FIELD_SKIP,
022    
023    /**
024     * Indicate that the field represents a constant or global 
025     * variable.  It is expected that the java field will be declared
026     * static.
027     */
028    CONSTANT,
029    
030    /**
031     * Indicate that the field is a pointer.
032     */
033    POINTER_FIELD,
034
035    /**
036     * Indicate that the getter method used is not part of
037     * the structure. Useful for using wrappers to access
038     * certain structure fields.
039     *
040     * Only useful when the getter is declared explicitly.
041     */
042    GETTER_NONMEMBER,
043
044    /**
045     * Indicate that the setter method used is not part of
046     * the structure. Useful for using wrappers to access
047     * certain structure fields.
048     *
049     * Only useful when the setter is declared explicitly.
050     */
051    SETTER_NONMEMBER,
052}
